```python
"""Stable identity and JSON persistence for a knowledge index."""
from __future__ import annotations

from dataclasses import dataclass, replace
import hashlib
import json
from pathlib import Path, PurePosixPath
import re
from typing import Any

_HASH_RE = re.compile(r"[0-9a-f]{64}")


class ManifestError(ValueError):
    """The index Manifest is missing, malformed, or internally inconsistent."""


def sha256_bytes(content: bytes) -> str:
    return hashlib.sha256(content).hexdigest()


@dataclass(frozen=True, order=True)
class CorpusFileManifest:
    path: str
    content_sha256: str
    chunk_count: int


@dataclass(frozen=True)
class IndexManifest:
    schema_version: int
    fingerprint: str
    collection_name: str
    built_at: str
    embedding_model: str
    embedding_dimensions: int
    chunk_size: int
    chunk_overlap: int
    corpus_root: str
    files: tuple[CorpusFileManifest, ...]

    @property
    def file_count(self) -> int:
        return len(self.files)

    @property
    def chunk_count(self) -> int:
        return sum(item.chunk_count for item in self.files)

    @classmethod
    def create(
        cls, *, schema_version: int, collection_name: str, built_at: str,
        embedding_model: str, embedding_dimensions: int, chunk_size: int,
        chunk_overlap: int, corpus_root: str,
        files: tuple[CorpusFileManifest, ...],
    ) -> "IndexManifest":
        manifest = cls(
            schema_version=schema_version, fingerprint="",
            collection_name=collection_name, built_at=built_at,
            embedding_model=embedding_model,
            embedding_dimensions=embedding_dimensions,
            chunk_size=chunk_size, chunk_overlap=chunk_overlap,
            corpus_root=corpus_root,
            files=tuple(sorted(files, key=lambda item: item.path)),
        )
        manifest = replace(manifest, fingerprint=manifest.compute_fingerprint())
        manifest.validate()
        return manifest

    def stable_payload(self) -> dict[str, Any]:
        return {
            "schema_version": self.schema_version,
            "collection_name": self.collection_name,
            "embedding": {
                "model": self.embedding_model,
                "dimensions": self.embedding_dimensions,
            },
            "chunking": {
                "chunk_size": self.chunk_size,
                "chunk_overlap": self.chunk_overlap,
            },
            "corpus": {
                "file_count": self.file_count,
                "chunk_count": self.chunk_count,
                "files": [
                    {
                        "path": item.path,
                        "content_sha256": item.content_sha256,
                        "chunk_count": item.chunk_count,
                    }
                    for item in self.files
                ],
            },
        }

    def compute_fingerprint(self) -> str:
        canonical = json.dumps(
            self.stable_payload(), ensure_ascii=False, sort_keys=True,
            separators=(",", ":"),
        ).encode("utf-8")
        return f"sha256:{sha256_bytes(canonical)}"

    def validate(self) -> None:
        if self.schema_version != 1:
            raise ManifestError("schema_version must be 1")
        if not self.collection_name.strip():
            raise ManifestError("collection_name must not be empty")
        if not self.embedding_model.strip():
            raise ManifestError("embedding model must not be empty")
        if self.embedding_dimensions <= 0:
            raise ManifestError("embedding dimensions must be positive")
        if self.chunk_size <= 0 or not 0 <= self.chunk_overlap < self.chunk_size:
            raise ManifestError("chunking requires 0 <= overlap < positive size")
        if not self.files:
            raise ManifestError("corpus files must not be empty")
        if list(self.files) != sorted(self.files, key=lambda item: item.path):
            raise ManifestError("corpus files must be sorted by path")
        for item in self.files:
            path = PurePosixPath(item.path)
            if (not item.path or "\\" in item.path or path.is_absolute()
                    or ".." in path.parts or path.as_posix() != item.path):
                raise ManifestError(f"invalid POSIX corpus path: {item.path}")
            if _HASH_RE.fullmatch(item.content_sha256) is None:
                raise ManifestError(f"invalid content_sha256: {item.path}")
            if item.chunk_count < 0:
                raise ManifestError(f"negative chunk_count: {item.path}")
        if self.fingerprint != self.compute_fingerprint():
            raise ManifestError("fingerprint does not match stable payload")

    def to_dict(self) -> dict[str, Any]:
        payload = self.stable_payload()
        payload["fingerprint"] = self.fingerprint
        payload["built_at"] = self.built_at
        payload["corpus"]["root"] = self.corpus_root
        return payload

    @classmethod
    def from_dict(cls, raw: dict[str, Any]) -> "IndexManifest":
        try:
            embedding = raw["embedding"]
            chunking = raw["chunking"]
            corpus = raw["corpus"]
            raw_files = corpus["files"]
            files = tuple(
                CorpusFileManifest(
                    path=str(item["path"]),
                    content_sha256=str(item["content_sha256"]),
                    chunk_count=int(item["chunk_count"]),
                )
                for item in raw_files
            )
            manifest = cls(
                schema_version=int(raw["schema_version"]),
                fingerprint=str(raw["fingerprint"]),
                collection_name=str(raw["collection_name"]),
                built_at=str(raw["built_at"]),
                embedding_model=str(embedding["model"]),
                embedding_dimensions=int(embedding["dimensions"]),
                chunk_size=int(chunking["chunk_size"]),
                chunk_overlap=int(chunking["chunk_overlap"]),
                corpus_root=str(corpus["root"]),
                files=tuple(sorted(files, key=lambda item: item.path)),
            )
            declared_file_count = int(corpus["file_count"])
            declared_chunk_count = int(corpus["chunk_count"])
        except (KeyError, TypeError, ValueError) as exc:
            raise ManifestError(f"invalid Manifest shape: {exc}") from exc
        manifest.validate()
        if declared_file_count != manifest.file_count:
            raise ManifestError("declared file_count does not match files")
        if declared_chunk_count != manifest.chunk_count:
            raise ManifestError("declared chunk_count does not match files")
        return manifest


def write_manifest(path: Path, manifest: IndexManifest) -> None:
    manifest.validate()
    path.parent.mkdir(parents=True, exist_ok=True)
    temporary = path.with_name(f"{path.name}.tmp")
    temporary.write_text(
        json.dumps(manifest.to_dict(), ensure_ascii=False, indent=2) + "\n",
        encoding="utf-8",
    )
    temporary.replace(path)


def load_manifest(path: Path) -> IndexManifest:
    try:
        raw = json.loads(path.read_text(encoding="utf-8"))
    except (OSError, UnicodeError, json.JSONDecodeError) as exc:
        raise ManifestError(f"cannot read Manifest {path}: {exc}") from exc
    if not isinstance(raw, dict):
        raise ManifestError("Manifest root must be a JSON object")
    return IndexManifest.from_dict(raw)
```

The nested JSON shape, canonical serialization, exclusions of `built_at` and `corpus.root`, validation, declared-total checks, and atomic temporary-file replacement in this code are the schema-1 contract. Do not add environment values or API credentials.

```python
"""Shared deterministic builder for persistent and frozen knowledge indexes."""
from __future__ import annotations

from collections.abc import Callable
from dataclasses import dataclass
from datetime import datetime, timezone
from pathlib import Path
from typing import Protocol

from app.services.index_manifest import (
    CorpusFileManifest,
    IndexManifest,
    sha256_bytes,
)
from app.services.knowledge_chunker import KnowledgeChunk, chunk_markdown
from app.services.rag_settings import (
    CHUNK_OVERLAP,
    CHUNK_SIZE,
    EMBEDDING_BATCH_SIZE,
    KNOWLEDGE_COLLECTION_NAME,
)


class EmbeddingProvider(Protocol):
    def embed_texts(self, texts: list[str]) -> list[list[float]]:
        raise NotImplementedError


class IndexRepository(Protocol):
    def rebuild(
        self, chunks: list[KnowledgeChunk], embeddings: list[list[float]],
    ) -> int:
        raise NotImplementedError


@dataclass(frozen=True)
class IndexBuildResult:
    indexed_count: int
    manifest: IndexManifest


def build_knowledge_index(
    *,
    corpus_root: Path,
    source_dir: Path,
    corpus_label: str,
    repository: IndexRepository,
    embedding_client: EmbeddingProvider,
    embedding_model: str,
    chunk_size: int = CHUNK_SIZE,
    chunk_overlap: int = CHUNK_OVERLAP,
    batch_size: int = EMBEDDING_BATCH_SIZE,
    collection_name: str = KNOWLEDGE_COLLECTION_NAME,
    progress: Callable[[str, int], None] | None = None,
) -> IndexBuildResult:
    if batch_size <= 0:
        raise ValueError("embedding batch size must be positive")
    if not corpus_label.strip():
        raise ValueError("corpus label must not be empty")
    if not embedding_model.strip():
        raise ValueError("embedding model must not be empty")
    if not collection_name.strip():
        raise ValueError("collection name must not be empty")

    source_root = corpus_root / source_dir
    markdown_files = sorted(
        source_root.rglob("*.md"),
        key=lambda path: path.relative_to(corpus_root).as_posix(),
    )
    if not markdown_files:
        raise ValueError(f"no Markdown files found in {source_root}")

    chunks: list[KnowledgeChunk] = []
    file_manifests: list[CorpusFileManifest] = []
    for path in markdown_files:
        content = path.read_bytes()
        source = path.relative_to(corpus_root).as_posix()
        file_chunks = chunk_markdown(
            text=content.decode("utf-8"),
            source=source,
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
        )
        chunks.extend(file_chunks)
        file_manifests.append(
            CorpusFileManifest(
                path=source,
                content_sha256=sha256_bytes(content),
                chunk_count=len(file_chunks),
            )
        )
        if progress is not None:
            progress(source, len(file_chunks))

    if not chunks:
        raise ValueError("Markdown corpus produced zero chunks")

    embeddings: list[list[float]] = []
    for start in range(0, len(chunks), batch_size):
        batch = chunks[start : start + batch_size]
        batch_embeddings = embedding_client.embed_texts(
            [chunk.content for chunk in batch]
        )
        if len(batch_embeddings) != len(batch):
            raise ValueError("embedding count does not match chunk count")
        embeddings.extend(batch_embeddings)

    dimensions = len(embeddings[0])
    if dimensions <= 0 or any(len(vector) != dimensions for vector in embeddings):
        raise ValueError("embeddings must all have the same positive dimension")

    indexed_count = repository.rebuild(chunks, embeddings)
    if indexed_count != len(chunks):
        raise ValueError("repository count does not match chunk count")

    manifest = IndexManifest.create(
        schema_version=1,
        collection_name=collection_name,
        built_at=datetime.now(timezone.utc).isoformat(),
        embedding_model=embedding_model,
        embedding_dimensions=dimensions,
        chunk_size=chunk_size,
        chunk_overlap=chunk_overlap,
        corpus_root=corpus_label,
        files=tuple(file_manifests),
    )
    if manifest.chunk_count != indexed_count:
        raise ValueError("Manifest chunk count does not match repository count")
    return IndexBuildResult(indexed_count=indexed_count, manifest=manifest)
```

This ordering is intentional: all reads, chunking, Embedding calls, and vector-shape validation finish before `repository.rebuild()`. A provider failure therefore cannot clear a working index. The builder records zero-chunk Markdown files but rejects a corpus whose total chunk count is zero.
